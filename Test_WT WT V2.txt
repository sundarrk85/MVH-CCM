        function handleCSVFile(file) {
            var reader = new FileReader();
            reader.onload = function(e) {
                var contents = e.target.result;
                processCSV(contents);
            };
            reader.onerror = function() {
                alert('Unable to read the catalog.');
            };
            reader.readAsText(file);
        }

        function handleXLSXFile(file) {
            var reader = new FileReader();
            reader.onload = function(e) {
                var data = new Uint8Array(e.target.result);
                var workbook = XLSX.read(data, { type: 'array' });
                var firstSheetName = workbook.SheetNames[0];
                var worksheet = workbook.Sheets[firstSheetName];
                var csv = XLSX.utils.sheet_to_csv(worksheet);
                processCSV(csv);
            };
            reader.onerror = function() {
                alert('Unable to read the Catalog.');
            };
            reader.readAsArrayBuffer(file);
        }
        
        function handleZIPFile(file) {
            var reader = new FileReader();
            reader.onload = function(e) {
                JSZip.loadAsync(e.target.result).then(function(zip) {
                    // Process each file in the ZIP archive
                    var files = Object.values(zip.files);
                    files.forEach(function(zipFile) {
                        if (!zipFile.dir) {
                            zipFile.async('blob').then(function(blob) {
                                var fileName = zipFile.name;
                                if (fileName.toLowerCase().endsWith('.csv')) {
                                    handleCSVFile(blob);
                                } else if (fileName.toLowerCase().endsWith('.xlsx')) {
                                    handleXLSXFile(blob);
                                } else {
                                    alert('Unsupported file type inside ZIP: ' + fileName);
                                }
                            });
                        }
                    });
                }, function(err) {
                    alert('Unable to read the ZIP file.');
                });
            };
            reader.onerror = function() {
                alert('Unable to read the ZIP file.');
            };
            reader.readAsArrayBuffer(file);
        }

        function processCSV(contents) {
            if (contents.trim() === '') {
                alert('The catalog is empty. Please upload a catalog with at least one header and one row.');
                document.getElementById('file-name').textContent = 'No file chosen';
                this.value = ''; // Clear the file input
              submitButton.disabled = true;
                return;
            }
            var lines = contents.split('\n');
            var rowCount = lines.length;

            // Check if the last line is empty (sometimes an extra new line adds an empty row)
            if (lines[rowCount - 1].trim() === '') {
                rowCount -= 1;
            }

            if (rowCount < 2) {
               // document.getElementById('file-name').textContent = 'No file chosen';
               // this.value = ''; // Clear the file input
              //submitButton.disabled = true;
                alert('The catalog has less than 2 rows. Please upload a catalog with at least one header and one row.');
                return;
            }

            //alert('The catalog is valid');
            showModal('The uploaded catalog is valid');
            //submitButton.disabled = false;
        }

        function UploadFileToBlob(inputFile) {
    var reader = new FileReader();
    // Read the file as an ArrayBuffer for binary content
    reader.readAsArrayBuffer(inputFile);
 
    reader.onload = function () {
        document.getElementById("statusMessage").style.display = "block";
        var file = {};
        file.name = inputFile.name;
 
        // Convert ArrayBuffer to Base64
        var binary = '';
        var bytes = new Uint8Array(reader.result);
        var len = bytes.byteLength;
        for (var i = 0; i < len; i++) {
            binary += String.fromCharCode(bytes[i]);
        }
        file.contentBytes = btoa(binary); // Base64 encode the binary data
    

 
        var data = {};
        
        data["File Name"] = inputFile.name;
        data["File Content"] = file;
        data["BU"] = getQueryParameter("unit");
        data["CatalogName"] = getQueryParameter("name");
        data["CatalogNumber"] = getQueryParameter("number");
        data["LastUpdatedVersion"] = getQueryParameter("version");
        data["No.ofItems"] = getQueryParameter("items");
        data["SupplierEmail"] = getQueryParameter("SupplierEmail");
        data["SupplierAccountName"] = getQueryParameter("SupplierAccountName");
        data["BaselineCatalogueGUID"] = getQueryParameter("BaselineCatalogueGUID");

        var payload = {};
        payload.eventData = JSON.stringify(data);
        alert(JSON.stringify(data));
        
 
        // Send the payload via AJAX
        shell.ajaxSafePost({
            type: "POST",
            contentType: "application/json",
            url: _uploadImageFlowURL,
            data: JSON.stringify(payload),
            processData: false,
            global: false,
            success: function (response) {
                console.log('Success:', response);
                document.getElementById("statusMessage").style.display = "none";
                alert('The Catalog has been uploaded successfully!');
                document.getElementById('file-upload').value = '';
                document.getElementById('file-name').textContent = 'No file chosen';
                window.location.href = 'https://supp-catalog-mgmt.powerappsportals.com/';
            },
            error: function (error) {
                console.error('Error:', error);
                document.getElementById("statusMessage").style.display = "none";
                alert('Catalog upload failed. Please try again.');
            }
        });
    };
 
    reader.onerror = function (error) {
        console.error('Error: ', error);
    };
}

        function handleSubmit() {
            var selectedFile;
            var selectedFileType = document.querySelector('input[name="uploadFile"]:checked').value;

            switch (selectedFileType) {
                case 'yes':
                    selectedFile = document.getElementById('zip-file-upload').files[0];
                    if (selectedFile) {
                        UploadFileToBlob(selectedFile);
                    } else {
                        alert("Please select a ZIP file before submitting.");
                    }
                    break;
                case 'no':
                    selectedFile = document.getElementById('file-upload').files[0];
                    if (selectedFile) {
                        UploadFileToBlob(selectedFile);
                    } else {
                        alert("Please select a Catalog before submitting.");
                    }
                    break;
            }
        }

        function handleCancel() {
            var selectedFileType = document.querySelector('input[name="uploadFile"]:checked').value;

            switch (selectedFileType) {
                case 'yes':
                    document.getElementById('zip-file-upload').value = ''; // Reset ZIP file input
                    document.getElementById('zip-file-name').textContent = 'No ZIP file chosen'; // Reset ZIP file name display
                    break;
                case 'no':
                    document.getElementById('file-upload').value = ''; // Reset CSV/XLSX file input
                    document.getElementById('file-name').textContent = 'No file chosen'; // Reset file name display
                    break;
            }

            alert('Upload canceled.');
            window.location.href = 'https://supp-catalog-mgmt.powerappsportals.com/';
        }
