        document.getElementById('file-upload').addEventListener('change', function () {

            var file = this.files[0];

            var fileName = file ? file.name : '';
            document.getElementById('file-name').textContent = '';
            // Check file extension
            if (!fileName.toLowerCase().endsWith('.csv') && !fileName.toLowerCase().endsWith('.xlsx')) {
                alert('Please upload a CSV or Excel catalog only.');
                document.getElementById('file-name').textContent = '';
                this.value = ''; // Clear the file input
                //submitButton.disabled = true;
                return;
            }

            // Check file size
            var maxSizeInBytes = 10 * 1024 * 1024; // 10 MB in bytes
            if (file.size > maxSizeInBytes) {
                alert('Maximum catalog size allowed is 10 MB.');
                document.getElementById('file-name').textContent = '';
                this.value = ''; // Clear the file input
                //  submitButton.disabled = true;
                return;
            }

            if (fileName.toLowerCase().endsWith('.csv')) {
                handleCSVFile(file);
            } else if (fileName.toLowerCase().endsWith('.xlsx')) {
                handleXLSXFile(file);
            }
        });

        function handleCSVFile(file) {
            var reader = new FileReader();
            reader.onload = function (e) {
                var contents = e.target.result;
                processCSV(contents);
            };
            reader.onerror = function () {
                alert('We are unable to read your catalog, please make sure you are uploading valid catalog.');
            };
            reader.readAsText(file);
        }

        function handleXLSXFile(file) {
            var reader = new FileReader();
            reader.onload = function (e) {
                var data = new Uint8Array(e.target.result);
                var workbook = XLSX.read(data, { type: 'array' });
                var firstSheetName = workbook.SheetNames[0];
                var worksheet = workbook.Sheets[firstSheetName];
                var csv = XLSX.utils.sheet_to_csv(worksheet);
                processCSV(csv);
            };
            reader.onerror = function () {
                alert('We are unable to read your catalog, please make sure you are uploading valid catalog.');
            };
            reader.readAsArrayBuffer(file);
        }

        function processCSV(contents) {
            if (contents.trim() === '') {
                alert('You have uploaded empty catalog. Please make sure you upload a catalog with at least one header and one row.');
                document.getElementById('file-name').textContent = '';
                this.value = ''; // Clear the file input
                submitButton.disabled = true;
                return;
            }
            var lines = contents.split('\n');
            var rowCount = lines.length;

            // Check if the last line is empty (sometimes an extra new line adds an empty row)
            if (lines[rowCount - 1].trim() === '') {
                rowCount -= 1;
            }

            if (rowCount < 2) {
                document.getElementById('file-name').textContent = '';
                this.value = ''; // Clear the file input
                //submitButton.disabled = true;
                alert('The catalog has less than 2 rows. Please make sure you upload a catalog with at least one header and one row.');
                return;
            }

            //alert('The catalog is valid');

            //showModal('You have uploaded valid catalog.');
            //submitButton.disabled = false;
        }

        function UploadFileToPowerAutomate(inputFile) {

            var _uploadImageFlowURL2 = "{{ UploadZipFlowURL }}";
            
            document.getElementById("statusMessage").style.display = "block";
            var formData = new FormData();
            formData.append("BU", getQueryParameter("unit"));
            formData.append("fileName", inputFile.name);
            formData.append("fileContent", inputFile);
            formData.append("CatalogName", getQueryParameter("name"));
            formData.append("CatalogNumber", getQueryParameter("number"));
            formData.append("LastUpdatedVersion", getQueryParameter("version"));
            formData.append("No.ofItems", getQueryParameter("items"));
            formData.append("SupplierEmail", getQueryParameter("SupplierEmail"));
            formData.append("SupplierAccountName", getQueryParameter("SupplierAccountName"));
            formData.append("BaselineCatalogueGUID", getQueryParameter("BaselineCatalogueGUID"));



            // Send the payload via AJAX
            shell.ajaxSafePost({
                type: "POST",

                url: _uploadImageFlowURL2,
                data: formData,
                processData: false, // Prevent jQuery from processing the form data
                contentType: false, // Prevent jQuery from setting the content-type header
                success: function (response) {

                    console.log('Success:', response);
                    document.getElementById("statusMessage").style.display = "none";
                    alert('Your catalog has been uploaded successfully!');
                    window.location.href = '/Catalogue_History/';
                    document.getElementById('zip-file-upload').value = '';
                    document.getElementById('zip-file-name').textContent = '';

                },
                error: function (error) {

                    console.error('Error:', error);
                    document.getElementById("statusMessage").style.display = "none";
                    alert('Your catalog upload failed. Please try again or contact SCM IT Support Team.');
                    //window.location.href = '/';
                }
            });

        }

        function UploadFileToBlob(inputFile) {
            var reader = new FileReader();

            reader.readAsDataURL(inputFile);

            reader.onload = function () {
                document.getElementById("statusMessage").style.display = "block";
                var file = {
                }
                    ;
                file.name = inputFile.name;
                file.contentBytes = reader.result.split(',')[1];

                var data = {
            };
                data["File Name"] = inputFile.name;
                data["File Content"] = file;
                data["BU"] = getQueryParameter("unit");
                data["CatalogName"] = getQueryParameter("name");
                data["CatalogNumber"] = getQueryParameter("number");
                data["LastUpdatedVersion"] = getQueryParameter("version");
                data["No.ofItems"] = getQueryParameter("items");
                data["SupplierEmail"] = getQueryParameter("SupplierEmail");
                data["SupplierAccountName"] = getQueryParameter("SupplierAccountName");
                data["BaselineCatalogueGUID"] = getQueryParameter("BaselineCatalogueGUID");
                var payload = {};
                payload.eventData = JSON.stringify(data);

                shell.ajaxSafePost({
                    type: "POST", contentType: "application/json",
                    url: _uploadImageFlowURL,
                    data: JSON.stringify(payload),
                    processData: false,
                    global: false,
                    success: function (response) {
                        //console.log('Success:', response);
                        document.getElementById("statusMessage").style.display = "none";
                        alert('Your catalog has been uploaded successfully!');
                        window.location.href = '/Catalogue_History/';
                        document.getElementById('file-upload').value = '';
                        document.getElementById('file-name').textContent = '';
                    },
                    error: function (error) {

                        document.getElementById("statusMessage").style.display = "none";
                        alert('Your catalog upload failed. Please try again or contact SCM IT Support Team.');
                        console.error(error);
                        window.location.href = '/';
                    },
                }
                )
                    .then(response => response.json())
                    .catch(error => {
                        console.error('Error:', error);
                        document.getElementById("statusMessage").style.display = "none";
                    });
            };
            reader.onerror = function (error) {
                console.error('Error: ', error);
            };
        }

        function handleSubmit(btn) {
            var selectedFile;
            var selectedFileType = document.querySelector('input[name="uploadFile"]:checked').value;

            switch (selectedFileType) {
                case 'yes':
                    selectedFile = document.getElementById('zip-file-upload').files[0];
                    if (selectedFile) {
                        btn.disabled = true;
                        UploadFileToPowerAutomate(selectedFile);

                    } else {
                        alert("Please select the valid ZIP file before submitting.");
                    }
                    break;
                case 'no':
                    selectedFile = document.getElementById('file-upload').files[0];
                    if (selectedFile) {
                        btn.disabled = true;
                        UploadFileToBlob(selectedFile);
                    } else {
                        alert("Please select the valid catalog before submitting.");
                    }
                    break;
            }
        }

        function handleCancel() {

            switch (selectedFileType) {
                case 'yes':
                    document.getElementById('zip-file-upload').value = ''; // Reset ZIP file input
                    document.getElementById('zip-file-name').textContent = 'No ZIP file chosen'; // Reset ZIP file name display
                    break;
                case 'no':
                    document.getElementById('file-upload').value = ''; // Reset CSV/XLSX file input
                    document.getElementById('file-name').textContent = ''; // Reset file name display
                    break;
            }


        }
